# include <stdio.h>
# include <malloc.h>

void f(int * q)
{
	//*p = 200; //error
	//q = 200;
	//**q = 200; //error
	*q = 200;
	//free(q); //把q所指向的内存释放掉
}

int main(void)
{
	int * p = (int *)malloc(sizeof(int)); //sizeof(int)返回值是int所占的字节数
	*p = 10;

	printf("%d\n", *p); //10
	f(p);//p 是int *类型
	printf("%d\n", *p); //200
}
