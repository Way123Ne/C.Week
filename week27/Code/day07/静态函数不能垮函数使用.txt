# include <stdio.h>

void f(int ** q) //q是个指针变量，无论q是什么类型的指针变量，都只占4个字节
{
	int i = 5; 
	//*q等价于p   q和**q都不等价于p
	//*q = i; //error 因为*q = i; 等价于 p = i;这样写是错误的 
	*q = &i; //p = &i;
}

int main(void)
{
	int *p;  //p保存i的地址，但不能访问i的空间，因为i是f()函数中的静态变量，f()函数一旦执行完毕，静态变量i将会被系统自动释放

	f(&p);
	printf("%d\n", *p); //本句与语法没有问题，但逻辑上有问题

	return 0;
}
