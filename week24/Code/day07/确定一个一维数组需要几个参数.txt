# include <stdio.h>

//f函数可以输出任何一个一维数组的内容
void f(int * pArr, int len)
{
	int i;

	for (i=0; i<len; ++i)
		printf("%d ", *(pArr+i)); //*pArr *(pArr+1) *(pArr+2)
	printf("\n");

}

int main(void)
{
	int a[5] = {1,3,5,6,8};
	int b[7] = {4,6,7,2,34,75,76};
	int c[100] = {1,222,333,462};

	f(a,5);//a 是 int *
	f(b,7);
	f(c,100);

	return 0;
}
