# include <stdio.h>

int main(void)
{
	int * p;//p是变量的名字，int * 表示p变量存放的是int类型变量的地址
			//int * p;不表示定义了一个名字叫做*p的变量
			//int * p;应该理解为：p是变量名，p变量的数据类型是 int * 类型
			//所谓int * 类型  实际就是存放int 类型变量的地址
	int i = 3;

	p = &i;
			/*
				1> p保存了i的地址，因此p指向i
				2> p不是i,i也不是p,更准确说：修改p的值不影响i的值，反推成立
				3> 如果一个指针变量指向了某个普通变量，则
						*指针变量   就完全等同于   普通变量
					例子：*p = i;
			*/

	* p = i;

	printf("p=%d\n",*p);

	return 0;
}
