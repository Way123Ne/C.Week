# include <stdio.h>

int main(void)
{
		printf("%d %d %d %d %d %d\n",3%3,13%-3,-13%3,-13%3,-13%23,3%5);
		return 0;
}
/*
	输出结果是：

*******************************************

0 1 -1 -1 -13 3

Press any key to continue
*******************************************

总结：取余%的运算对象必须是整数，结果是整除后的余数，其余数的符号与被除数相同
*/
