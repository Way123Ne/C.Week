# include <stdio.h>

int main(void)
{
	if (3 > 2)
		printf("AAAA\n");
	else if (3 > 1)
		printf("BBBB\n");
	else
		printf("CCCC\n");

	return 0;
}

/*
--------------------------------
--------------------------------
总结:
if (表达式1)
	A;
else if (表达式2)
	B;
else if (表达式3)
	C;
else
	D;
